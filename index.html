Creating a text element

To use PyGameUI, you need to start with a basic pygame script:

            
                import pygame, sys, PyGameUI 



                pygame.init()



                win = pygame.display.set_mode((500, 500))



                clock = pygame.time.Clock()

                while True:

                    win.fill((0, 0, 0)) 

                    for event in pygame.event.get():

                        if event.type == pygame.QUIT:

                            pygame.quit()

                            sys.exit()



                    pygame.display.flip()

                    clock.tick(60)

Adding a text element only requires 1 line of code:
The three arguments are position (x, y), content and color (r, g, b).

            
                example_text = PyGameUI.text((200, 200), "My cool text", (255, 255, 255))

Drawing the text only requires 1 line of code too: 

            
                example_text.draw(win)

Final code:

            
                import pygame, sys, PyGameUI 



                pygame.init()



                win = pygame.display.set_mode((500, 500))



                # Create text element

                example_text = PyGameUI.text((200, 200), "My cool text", (255, 255, 255))



                clock = pygame.time.Clock()

                while True:

                    win.fill((0, 0, 0))

                    for event in pygame.event.get():

                        if event.type == pygame.QUIT:

                            pygame.quit()

                            sys.exit()


                    
                    # Draw text 

                    example_text.draw(win)



                    pygame.display.flip()

                    clock.tick(60)
